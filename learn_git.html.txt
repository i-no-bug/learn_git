<script>
    // 用户发送消息的功能
    function sendMessage() {
        const userInput = document.getElementById('user-input'); // 获取输入框
        const chatBody = document.getElementById('chat-body'); // 获取聊天内容区域

        // 如果输入框内容为空，不执行操作
        if (userInput.value.trim() === "") return;

        // 创建用户消息
        const userMessage = document.createElement('div');
        userMessage.classList.add('message', 'user'); // 添加样式类
        userMessage.innerHTML = `<span class="content">${userInput.value}</span>`;
        chatBody.appendChild(userMessage); // 添加到聊天内容区域

        // 创建客服自动回复消息
        const botMessage = document.createElement('div');
        botMessage.classList.add('message', 'bot'); // 添加样式类
        botMessage.innerHTML = `<span class="content">抱歉，我暂时无法理解您的问题。</span>`;
        chatBody.appendChild(botMessage); // 添加到聊天内容区域

        // 清空输入框
        userInput.value = "";

        // 滚动到最新消息
        chatBody.scrollTop = chatBody.scrollHeight;
    }
</script>
